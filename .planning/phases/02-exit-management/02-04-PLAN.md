# File: verify_phase2.py (run from BetterBot root, do not commit)
import sys
from datetime import datetime

# ---- Import ----
from Scripts.exit_manager import ExitManager, EntryRecord
from Scripts.config import (
    HARD_STOP_ATR_MULT, TAKE_PROFIT_ATR_MULT,
    TRAILING_ATR_MULT, TRAILING_TRIGGER_ATR, DAILY_LOSS_LIMIT_PCT
)

print("=== Phase 2 ExitManager Logic Tests ===")

# EXIT-01: Hard stop loss = entry_price - (2 * ATR) for long
em = ExitManager()
em.register_entry('TEST', 'long', 100.0, 2.0, 10)
rec = em.positions['TEST']
assert rec.stop_price == 96.0, f"EXIT-01 FAIL: stop={rec.stop_price}, expected 96.0"
print("EXIT-01 PASS: stop_price = entry - 2*ATR = 96.0")

# EXIT-02: Take profit = entry_price + (4 * ATR) for long
assert rec.target_price == 108.0, f"EXIT-02 FAIL: target={rec.target_price}, expected 108.0"
print("EXIT-02 PASS: target_price = entry + 4*ATR = 108.0")

# EXIT-01 short direction: stop = entry + 2*ATR
em2 = ExitManager()
em2.register_entry('SHORT_TEST', 'short', 100.0, 2.0, 10)
rec2 = em2.positions['SHORT_TEST']
assert rec2.stop_price == 104.0, f"EXIT-01 short FAIL: stop={rec2.stop_price}, expected 104.0"
print("EXIT-01 short PASS: stop_price = entry + 2*ATR = 104.0")

# EXIT-03: Trailing stop activates after 1x ATR profit, trails at 1.5x ATR below high
em3 = ExitManager()
em3.register_entry('TRAIL_TEST', 'long', 100.0, 2.0, 10)
# Price moves up but < 1x ATR profit — trailing stop should NOT activate
result = em3.check_exits('TRAIL_TEST', 101.5)  # profit=1.5 < 1x ATR=2.0? Wait 2.0*1.0=2.0, 1.5 < 2.0
assert em3.positions['TRAIL_TEST'].trailing_stop is None, "EXIT-03 FAIL: trailing activated too early"
print("EXIT-03 PASS: trailing stop not activated at 1.5 profit (threshold is 2.0)")
# Price moves up >= 1x ATR profit (profit=2.5 >= trigger=2.0)
result = em3.check_exits('TRAIL_TEST', 102.5)  # profit=2.5 >= 2.0 trigger
assert em3.positions['TRAIL_TEST'].trailing_stop is not None, "EXIT-03 FAIL: trailing not activated at 2.5 profit"
trail = em3.positions['TRAIL_TEST'].trailing_stop
expected_trail = 102.5 - (1.5 * 2.0)  # high=102.5, mult=1.5, atr=2.0 → 99.5
assert trail == expected_trail, f"EXIT-03 FAIL: trail={trail}, expected {expected_trail}"
print(f"EXIT-03 PASS: trailing stop activated at {trail} (high=102.5 - 1.5*2.0=99.5)")

# EXIT-03: TRAILING_STOP return when price falls below trailing stop
result = em3.check_exits('TRAIL_TEST', 99.0)  # below trail=99.5
assert result == 'TRAILING_STOP', f"EXIT-03 FAIL: expected TRAILING_STOP, got {result}"
print("EXIT-03 PASS: TRAILING_STOP triggered when price falls below trailing stop")

# EXIT-01: HARD_STOP_LOSS return when price hits hard stop
em4 = ExitManager()
em4.register_entry('STOP_TEST', 'long', 100.0, 2.0, 10)
result = em4.check_exits('STOP_TEST', 95.9)  # below stop=96.0
assert result == 'HARD_STOP_LOSS', f"EXIT-01 FAIL: expected HARD_STOP_LOSS, got {result}"
print("EXIT-01 PASS: HARD_STOP_LOSS triggered when price falls to stop level")

# EXIT-02: TAKE_PROFIT return when price hits target
result = em4.check_exits('STOP_TEST', 108.0)  # should not reach here if stopped, but test independently
em5 = ExitManager()
em5.register_entry('TP_TEST', 'long', 100.0, 2.0, 10)
result = em5.check_exits('TP_TEST', 108.1)
assert result == 'TAKE_PROFIT', f"EXIT-02 FAIL: expected TAKE_PROFIT, got {result}"
print("EXIT-02 PASS: TAKE_PROFIT triggered at target price")

# EXIT-05: Circuit breaker activates at -2% NAV
em6 = ExitManager()
assert not em6.is_circuit_breaker_active(), "EXIT-05 FAIL: circuit breaker active on fresh instance"
em6.record_trade_pnl(-200.0, 9000.0)  # -200 / 9000 = -2.22% < -2%
assert em6.is_circuit_breaker_active(), "EXIT-05 FAIL: circuit breaker did not activate at -2.22%"
print("EXIT-05 PASS: circuit breaker activates when daily P&L breaches -2% NAV")

# EXIT-05: Circuit breaker does NOT suppress exits (exits run before circuit breaker check)
em7 = ExitManager()
em7.register_entry('CB_TEST', 'long', 100.0, 2.0, 10)
em7.record_trade_pnl(-300.0, 5000.0)  # trigger circuit breaker
assert em7.is_circuit_breaker_active()
# check_exits should still return a signal even when circuit breaker is active
result = em7.check_exits('CB_TEST', 95.9)  # below stop
assert result == 'HARD_STOP_LOSS', f"EXIT-05 FAIL: exits suppressed by circuit breaker, got {result}"
print("EXIT-05 PASS: check_exits unaffected by circuit breaker state")

# EXIT-05: Circuit breaker resets on a new day
em6._circuit_breaker_date = '2000-01-01'  # simulate yesterday
assert not em6.is_circuit_breaker_active(), "EXIT-05 FAIL: circuit breaker did not reset on new day"
print("EXIT-05 PASS: circuit breaker resets on new calendar day")

# ATR NaN guard
em8 = ExitManager()
em8.register_entry('NAN_TEST', 'long', 100.0, float('nan'), 5)
rec8 = em8.positions['NAN_TEST']
assert rec8.stop_price is not None and not (rec8.stop_price != rec8.stop_price), \
    f"ATR NaN guard FAIL: stop={rec8.stop_price}"
print(f"ATR NaN guard PASS: fallback atr=0.5%, stop={rec8.stop_price:.4f}")

print()
print("=== All EXIT logic assertions PASSED ===")

import inspect

print("=== Phase 2 Schema and Wiring Tests ===")

# DB models
from Scripts.database import PositionRegistry, TradeLog, Base
pr_cols = {c.name for c in PositionRegistry.__table__.columns}
required_pr = {'symbol', 'direction', 'entry_price', 'entry_time', 'atr_at_entry',
               'quantity', 'stop_price', 'target_price', 'trailing_high', 'trailing_stop'}
assert required_pr.issubset(pr_cols), f"PositionRegistry missing columns: {required_pr - pr_cols}"
print("PositionRegistry schema PASS:", sorted(pr_cols))

tl_cols = {c.name for c in TradeLog.__table__.columns}
required_tl = {'symbol', 'action', 'exit_reason', 'entry_reason', 'regime_at_decision',
               'sentiment_score', 'prediction_confidence', 'stop_price', 'target_price', 'decision_time'}
assert required_tl.issubset(tl_cols), f"TradeLog missing columns: {required_tl - tl_cols}"
print("TradeLog schema PASS (OBS-01):", sorted(tl_cols))

# PositionRegistry unique constraint on symbol
from sqlalchemy import UniqueConstraint
constraints = [c for c in PositionRegistry.__table__.constraints if isinstance(c, UniqueConstraint)]
unique_cols = {col.name for uc in constraints for col in uc.columns}
assert 'symbol' in unique_cols or any(
    col.unique for col in PositionRegistry.__table__.columns if col.name == 'symbol'
), "PositionRegistry.symbol not unique — upsert will fail"
print("PositionRegistry.symbol unique constraint PASS")

# Config constants
from Scripts.config import (HARD_STOP_ATR_MULT, TAKE_PROFIT_ATR_MULT, TRAILING_ATR_MULT,
                             TRAILING_TRIGGER_ATR, DAILY_LOSS_LIMIT_PCT)

assert TAKE_PROFIT_ATR_MULT == 4.0
assert TRAILING_ATR_MULT == 1.5
